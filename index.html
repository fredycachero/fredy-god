// Estado del juego let currentPlayer = 'X'; let board = ['', '', '', '', '', '', '', '', '']; let gameActive = true; // Referencias del tablero y el botón const boardElement = document.getElementById('board'); const resetButton = document.getElementById('resetButton'); // Crear el tablero de juego function createBoard() { boardElement.innerHTML = ''; board.forEach((cell, index) => { const cellElement = document.createElement('div'); cellElement.classList.add('cell'); cellElement.textContent = cell; cellElement.dataset.index = index; cellElement.addEventListener('click', handleCellClick); boardElement.appendChild(cellElement); }); } // Lógica cuando se hace clic en una celda function handleCellClick(event) { const index = event.target.dataset.index; // Si la celda está ocupada o el juego terminó, no hacer nada if (board[index] !== '' || !gameActive) return; // Marca la celda con el jugador actual board[index] = currentPlayer; event.target.textContent = currentPlayer; // Verifica si hay un ganador if (checkWinner()) { alert(`${currentPlayer} ha ganado!`); gameActive = false; return; } // Cambia al siguiente jugador currentPlayer = currentPlayer === 'X' ? 'O' : 'X'; // Verifica si el tablero está lleno (empate) if (!board.includes('')) { alert('¡Es un empate!'); gameActive = false; } } // Verifica si hay un ganador function checkWinner() { const winPatterns = [ [0, 1, 2], [3, 4, 5], [6, 7, 8], // Filas [0, 3, 6], [1, 4, 7], [2, 5, 8], // Columnas [0, 4, 8], [2, 4, 6] // Diagonales ]; for (let pattern of winPatterns) { const [a, b, c] = pattern; if (board[a] && board[a] === board[b] && board[a] === board[c]) { return true; } } return false; } // Reiniciar el juego resetButton.addEventListener('click', () => { board = ['', '', '', '', '', '', '', '', '']; currentPlayer = 'X'; gameActive = true; createBoard(); }); // Inicializa el juego createBoard();
