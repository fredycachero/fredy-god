// Lista de números para las cartas const cardValues = [1, 2, 3, 4, 5, 6, 7, 8]; const cards = [...cardValues, ...cardValues]; // Duplicamos los valores para emparejarlos // Mezclar las cartas cards.sort(() => Math.random() - 0.5); let flippedCards = []; let matchedCards = 0; const totalCards = cards.length; const gameBoard = document.getElementById('gameBoard'); // Crear las cartas cards.forEach((value, index) => { const card = document.createElement('div'); card.classList.add('card'); card.dataset.value = value; card.dataset.index = index; card.addEventListener('click', flipCard); gameBoard.appendChild(card); }); // Función para voltear las cartas function flipCard() { const card = this; if (flippedCards.length < 2 && !card.classList.contains('flipped')) { card.classList.add('flipped'); card.textContent = card.dataset.value; flippedCards.push(card); if (flippedCards.length === 2) { setTimeout(checkMatch, 1000); } } } // Comprobar si las dos cartas coinciden function checkMatch() { const [card1, card2] = flippedCards; if (card1.dataset.value === card2.dataset.value) { matchedCards += 2; if (matchedCards === totalCards) { alert('¡Felicidades! Has ganado el juego.'); } } else { card1.classList.remove('flipped'); card2.classList.remove('flipped'); card1.textContent = ''; card2.textContent = ''; } flippedCards = []; } // Botón para reiniciar el juego document.getElementById('resetButton').addEventListener('click', () => { matchedCards = 0; flippedCards = []; gameBoard.innerHTML = ''; cards.sort(() => Math.random() - 0.5); cards.forEach((value, index) => { const card = document.createElement('div'); card.classList.add('card'); card.dataset.value = value; card.dataset.index = index; card.addEventListener('click', flipCard); gameBoard.appendChild(card); }); });
